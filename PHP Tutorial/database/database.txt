7. USE blog;
8. CREATE TABLE users(
	id INT AUTO_INCREMENT,
	first_name varchar(50) NOT NULL,
	last_name varchar(50) NOT NULL,
	email_address varchar(100) NOT NULL,
	PRIMARY KEY (id)
);
9. show tables;
10. describe users;
11. SELECT *FROM users;
12. Now Insert the data row y row
	INSERT INTO users VALUES(null,'FirstName', 'LastName', 'EmailAddress');
	or
	INSERT INTO users('first_name', 'last_name', 'email_address') VALUES('FirstName', 'LastName', 'EmailAddress');
13. Show the row value
	SELECT *FROM users;
	or
	SELECT *FROM users ORDER BY name ASC/DESC;
	or
	SELECT *FROM users LIMIT 2;
	or
	/* Using Alais combine multiple column*/
	SELECT CONCAT_WS(' ', first_name, last_name) AS 'Full Name', email_address AS 'Email' FROM users;
	
14. Delete row data
	DELETE *FROM users WHERE first_name = 'FirstName';

15. Create another table name posts

	CREATE TABLE posts(
		id INT AUTO_INCREMENT,
		title varchar(150),
		body text,
		author_id INT NOT NULL,
		PRIMARY KEY(id)
	);
	
16. Insert data
	INSERT INTO posts(title, body, author_id) VALUES('My first post', 'The body of my post', 1);
	
17. UPDATE posts SET author_id = 2 WHERE id = 1;
	UPDATE posts SET title='Update Again', body='Update Body' WHERE id=1;

20. Joining Tables
	SELECT *FROM users INNER JOIN orders ON orders.user_id = users.id;
	### Another system
	SELECT users.id, first_name, order_number AS number FROM users INNER JOIN orders ON orders.user_id = users.id WHERE users.id = 1;
	## Count Total Order
	SELECT users.id, first_name, COUNT(order_number) AS number FROM users INNER JOIN orders ON orders.user_id = users.id WHERE users.id = 1;
